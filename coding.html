Database Backend (SQLite)

Python SQLite Backend

A lightweight SQLite database layer used to store account information and transaction history. This module handles table creation, inserts, updates, and queries, providing a clean backend interface for the BankAccount class. 


class Database:
    def __init__(self):
        self._connection = sqlite3.connect('Bankaccts.db')
        self._cursor = self._connection.cursor()
        self.create_tables()

    def create_tables(self):
        self.cursor.execute('CREATE TABLE IF NOT EXISTS accounts('
                            'acct_num integer PRIMARY KEY,'
                            'first_name varchar(50), last_name varchar(50),'
                            'acct_balance real)')
        
        self.cursor.execute('CREATE TABLE IF NOT EXISTS transactions('
                            'trans_id integer PRIMARY KEY,'
                            'trans_type varchar(10), trans_date varchar(20),'
                            'acct_number integer, trans_amount real)')
        
        self.connection.commit()

Grade Analysis Program

Python File I/O Menu System

A menu-driven Python program that reads grades from a CSV file and calculates the highest, lowest, and average values. This example demonstrates file handling, error checking, data parsing, and modular function design. 


def highestGrade():
    try:
        fin = open("exam_grades.csv",'r', encoding='utf-8')
    except FileNotFoundError:
        print("Unable to open file. Program will terminate.")
        return
    contents = fin.read()
    fin.close()
    lstGrades = [float(x) for x in contents.split(",")]
    print(f"The Highest Grade Is: {max(lstGrades)}")

def lowestGrade():
    try:
        fin = open("exam_grades.csv",'r', encoding='utf-8')
    except FileNotFoundError:
        print("Unable to open file. Program will terminate.")
        return
    contents = fin.read()
    fin.close()
    lstGrades = [float(x) for x in contents.split(",")]
    print(f"The Lowest Grade Is: {min(lstGrades)}")

def averageGrade():
    try:
        fin = open("exam_grades.csv",'r', encoding='utf-8')
    except FileNotFoundError:
        print("Unable to open file. Program will terminate.")
        return
    contents = fin.read()
    fin.close()
    lstGrades = [float(x) for x in contents.split(",")]
    print(f"The Average Grade Is: {sum(lstGrades)/len(lstGrades):.2f}")

def printMenu():
    menu = ('Enter a number:\n'
            ' (1) Get Highest Grade\n'
            ' (2) Get Lowest Grade\n'
            ' (3) Get Average Grade\n'
            ' Choose an option (1-3): ')
    try:
        return int(input(menu))
    except ValueError:
        return False

def driver():
    selection = printMenu()
    if selection == False:
        print('Error reading value from the user. Program will terminate.')
        return
    elif selection == 1:
        highestGrade()
    elif selection == 2:
        lowestGrade()
    elif selection == 3:
        averageGrade()

Telecom Analytics SQL Queries

SQL Joins Aggregation

A collection of SQL queries written for a telecom-style database. These examples demonstrate multi-table joins, grouping, aggregation, HAVING filters, and full dataset reconstruction. They highlight practical SQL skills used in reporting, analytics, and backend data processing. 


-- Query A: Active subscriptions
SELECT C.CustomerName, S.Status, S.StartDate
FROM fp.LLCUSTOMER AS C
JOIN fp.LLSUBSCRIPTION AS S ON C.CustomerID = S.CustomerID
WHERE S.Status = 'Active'
ORDER BY S.StartDate;

-- Query B: Total usage by activity type
SELECT C.CustomerName, CO.ActivityType, SUM(CO.Quantity) AS TotalUse,
       P.Minutes, P.DataGB, P.Texts
FROM fp.LLCUSTOMER AS C
JOIN fp.LLSUBSCRIPTION AS S ON C.CustomerID = S.CustomerID
JOIN fp.LLMOBILEPLAN AS P ON P.PlanName = S.PlanName
JOIN fp.LLCOMMS AS CO ON S.SubscriptionID = CO.SubscriptionID
GROUP BY C.CustomerName, CO.ActivityType, P.Minutes, P.DataGB, P.Texts
ORDER BY C.CustomerName, CO.ActivityType;

-- Query C: Customers exceeding plan limits (Voice example)
SELECT C.CustomerName, CO.ActivityType, SUM(CO.Quantity) AS TotalUse, P.Minutes
FROM fp.LLCUSTOMER AS C
JOIN fp.LLSUBSCRIPTION AS S ON C.CustomerID = S.CustomerID
JOIN fp.LLMOBILEPLAN AS P ON P.PlanName = S.PlanName
JOIN fp.LLCOMMS AS CO ON S.SubscriptionID = CO.SubscriptionID
WHERE CO.ActivityType = 'Voice'
GROUP BY C.CustomerName, CO.ActivityType, P.Minutes
HAVING SUM(CO.Quantity) > P.Minutes;

-- Query D: Recreate full spreadsheet
SELECT
    C.CustomerName AS 'Customer Name',
    C.Address AS Address,
    C.City AS City,
    C.PostalCode AS 'Postal Code',
    C.EmailAddress AS 'Email Address',
    S.Status AS 'Subscription Status',
    S.Discount AS 'Customer Discount',
    S.StartDate AS 'Subscription Start Date',
    P.PlanName AS 'Plan Name',
    P.Minutes AS 'Voice Minutes',
    P.DataGB AS 'Data GB',
    P.Texts AS Texts,
    P.Cost AS 'Plan Cost',
    CO.UsageDateTime AS 'Usage Date/Time',
    CO.ActivityType AS Activity,
    CO.Quantity AS Quantity
FROM fp.LLCUSTOMER AS C
JOIN fp.LLSUBSCRIPTION AS S ON C.CustomerID = S.CustomerID
JOIN fp.LLMOBILEPLAN AS P ON P.PlanName = S.PlanName
JOIN fp.LLCOMMS AS CO ON S.SubscriptionID = CO.SubscriptionID
ORDER BY C.CustomerID, S.SubscriptionID, CO.CommID;
